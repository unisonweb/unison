-- | for support of if .. then .. else
ifThenElse :: Prelude.Bool -> a -> a -> a
ifThenElse :: forall a. Bool -> a -> a -> a
ifThenElse Bool
Prelude.True  a
e1 a
_  = a
e1
ifThenElse Bool
Prelude.False a
_  a
e2 = a
e2
