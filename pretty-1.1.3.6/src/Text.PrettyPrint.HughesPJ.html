-- | Above, except that if the last line of the first argument stops
-- at least one position before the first line of the second begins,
-- these two lines are overlapped.  For example:
--
-- >    text "hi" $$ nest 5 (text "there")

-- lays out as
--
-- >    hi   there
--
-- rather than
--
-- >    hi
-- >         there
