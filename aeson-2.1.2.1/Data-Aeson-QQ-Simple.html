aesonQQ :: QuasiQuoter Source #

Converts a string representation of a JSON value into Value at compile-time.

{-# LANGUAGE QuasiQuotes #-}

import Data.Aeson (Value)
import Data.Aeson.QQ.Simple

joe :: Value
joe = [aesonQQ|{ "name": "Joe", "age": 12 }|]
