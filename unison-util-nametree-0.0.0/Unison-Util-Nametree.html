Flattening and unflattening

flattenNametree :: forall a b. Ord b => (a -> Map NameSegment b) -> Nametree a -> BiMultimap b Name Source #

flattenNametree organizes a nametree like

"foo" = #foo
"foo": {
  "bar" = #bar
  "bar": {
    "baz" = #baz
  }
}

into an equivalent-but-flatter association between names and definitions, like

{
  "foo" = #bar,
  "foo.bar" = #bar,
  "foo.bar.baz" = #baz
}

unflattenNametree :: Ord a => BiMultimap a Name -> Nametree (Map NameSegment a) Source #

unflattenNametree organizes an association between names and definitions like

{
  "foo" = #bar,
  "foo.bar" = #bar,
  "foo.bar.baz" = #baz
}

into an equivalent-but-less-flat nametree, like

"foo" = #foo
"foo": {
  "bar" = #bar
  "bar": {
    "baz" = #baz
  }
}
