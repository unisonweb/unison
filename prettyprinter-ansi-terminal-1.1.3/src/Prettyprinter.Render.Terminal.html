-- | Render 'SimpleDocStream' in a terminal.

    -- * Styling
    AnsiStyle,
    Color(..),

    -- ** Font color
    color, colorDull,

    -- ** Background color
    bgColor, bgColorDull,

    -- ** Font style
    bold, italicized, underlined,

    Bold(..),
    Underlined(..),
    Italicized(..),

    -- * Conversion to ANSI-infused 'Text'
    renderLazy, renderStrict,

    renderIO,

    -- ** Convenience functions
    putDoc, hPutDoc,
) where
