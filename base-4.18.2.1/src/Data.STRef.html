-- Be warned that 'modifySTRef' does not apply the function strictly.  This

--
-- >>> import Control.Monad (replicateM_)

--     ref <- newSTRef 0
--     replicateM_ 1000 $ modifySTRef ref (+1)
--     readSTRef ref ))
-- :}
-- 1000
--
